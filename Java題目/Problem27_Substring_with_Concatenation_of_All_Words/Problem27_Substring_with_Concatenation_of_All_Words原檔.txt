class Solution {
    public List<Integer> findSubstring(String s, String[] words) {

        List<Integer> numberList = new ArrayList<>();
        List<String> wordList = new ArrayList<>();
        Order(words,wordList,0);
        int length = wordList.get(0).length();
        for(String i : wordList){
            if(!s.equals(s.replace(i,""))){
                for(int j=0; j<s.length()-length+1; j++){
                    if(s.substring(j,j+length).equals(i)){
                        if(!numberList.contains(j)){numberList.add(j);}}
                }


            }
        }



        return numberList;
    }

    public  void Order(String[] words, List<String> list, int layer){


        //出口
        if(layer == words.length-1){
            StringBuilder temp = new StringBuilder();
            for(String i: words){temp.append(i);}
            list.add(temp.toString());
            }

        //動作
        for(int i=layer; i<words.length; i++){
            String temp = words[layer];
            words[layer] = words[i];
            words[i] = temp;
            Order(words,list,layer+1);
            words[i] = words[layer];
            words[layer] = temp;
        }



    }




}
