//解法：以DP方式得出1~n的解，套用公式f(n) = f(0)*f(n-1) + f(1)*f(n-2) + ...... + f(n-1)*f(0)
class Solution {
    public int numTrees(int n) {

        int[] dp = new int[n+1];
        dp[0] = 1;
        for(int i=1; i<=n; i++){
            for(int j=0; j<i; j++){
                dp[i] += dp[j]*dp[i-j-1];
            }
        }
        return dp[n];
    }

}
